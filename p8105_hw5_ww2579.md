tidy = function(y) {
  pivot_longer(
    data = y,
    week_1:week_8,
    names_to = "week",

``` r
n = 30
mu = 0
sigma = 5
set.seed(1)


sim = function(n = 30, mu, sigma = 5) {
  samp = tibble(
    x = rnorm(n, mean = mu, sd = sigma))
  samp %>%
    t.test(conf.level = 0.95, formula = samp~1) %>%
    broom::tidy()}

sim_results = tibble(diff_mu = c(0,1,2,3,4,5,6)) %>%
  mutate(output = 
           map(.x = diff_mu, ~rerun(5000, sim(mu = .x))))
```
